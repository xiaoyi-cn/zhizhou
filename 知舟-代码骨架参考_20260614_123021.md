## 5. 加密工具 ✅

路径：`server/internal/pkg/crypto/crypto.go`

```go
package crypto

import (
	"crypto/aes"
	"crypto/cipher"
	"crypto/rand"
	"encoding/base64"
	"errors"
	"io"
)

// Encrypt 使用 AES-GCM 加密明文，返回 base64 编码的密文
func Encrypt(key []byte, plainText string) (string, error) {
	block, err := aes.NewCipher(key)
	if err != nil {
		return "", err
	}

	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return "", err
	}

	nonce := make([]byte, gcm.NonceSize())
	if _, err = io.ReadFull(rand.Reader, nonce); err != nil {
		return "", err
	}

	// Seal 会把 nonce 和密文拼在一起
	cipherText := gcm.Seal(nonce, nonce, []byte(plainText), nil)
	return base64.StdEncoding.EncodeToString(cipherText), nil
}

// Decrypt 使用 AES-GCM 解密 base64 编码的密文
func Decrypt(key []byte, cryptoText string) (string, error) {
	data, err := base64.StdEncoding.DecodeString(cryptoText)
	if err != nil {
		return "", err
	}

	block, err := aes.NewCipher(key)
	if err != nil {
		return "", err
	}

	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return "", err
	}

	nonceSize := gcm.NonceSize()
	if len(data) < nonceSize {
		return "", errors.New("ciphertext too short")
	}

	nonce, encrypted := data[:nonceSize], data[nonceSize:]
	plainText, err := gcm.Open(nil, nonce, encrypted, nil)
	if err != nil {
		return "", err
	}

	return string(plainText), nil
}
```

> **使用注意**：key 必须是 32 字节（AES-256），从环境变量 `ENCRYPTION_KEY` 注入，不可硬编码。

package repository

import (
	"database/sql"

	"github.com/lib/pq"
	"github.com/your-username/zhizhou/server/internal/model"
)

type APIKeyRepository interface {
	Create(apiKey *model.APIKey) error
	GetByUserID(userID string) ([]model.APIKey, error)
}

type apiKeyRepo struct {
	db *sql.DB
}

func NewAPIKeyRepository(db *sql.DB) APIKeyRepository {
	return &apiKeyRepo{db: db}
}

func (r *apiKeyRepo) Create(k *model.APIKey) error {
	query := `INSERT INTO api_keys (user_id, provider, api_key_encrypted, base_url, model) 
			  VALUES ($1, $2, $3, $4, $5) RETURNING id`
	return r.db.QueryRow(query, k.UserID, k.Provider, k.APIKeyEncrypted, k.BaseURL, k.Model).Scan(&k.ID)
}

func (r *apiKeyRepo) GetByUserID(userID string) ([]model.APIKey, error) {
	rows, err := r.db.Query(`SELECT id, provider, api_key_encrypted, base_url, model, is_active 
	                        FROM api_keys WHERE user_id = $1`, userID)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var keys []model.APIKey
	for rows.Next() {
		var k model.APIKey
		err := rows.Scan(&k.ID, &k.Provider, &k.APIKeyEncrypted, &k.BaseURL, &k.Model, &k.IsActive)
		if err != nil {
			return nil, err
		}
		keys = append(keys, k)
	}
	return keys, nil
}
```

import (

type ContentRepository interface {
	Create(c *model.Content) error

type contentRepo struct {
	db *sql.DB
}

func NewContentRepository(db *sql.DB) ContentRepository {
	return &contentRepo{db: db}
}

// Create 插入新内容，使用 PostgreSQL 的 RETURNING 返回生成的 ID
func (r *contentRepo) Create(c *model.Content) error {
	query := `
		INSERT INTO contents (user_id, url, title, source_type, raw_content, status)
		VALUES ($1, $2, $3, $4, $5, $6)
		RETURNING id, created_at`
	
	return r.db.QueryRow(query, 
		c.UserID, 
		c.URL, 
		c.Title,        // 从 parser 获取的真实标题
		"web",
		c.RawContent,
		c.Status,
	).Scan(&c.ID, &c.CreatedAt)
}

// GetPendingByUserID 获取待审核的内容，利用索引优化查询性能
func (r *contentRepo) GetPendingByUserID(userID string) ([]model.Content, error) {
	query := `
		SELECT id, url, title, summary, category, tags, status, created_at 
		FROM contents 
		WHERE user_id = $1 AND status = 'pending'
		ORDER BY created_at DESC`
	
	rows, err := r.db.Query(query, userID)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var contents []model.Content
	for rows.Next() {
		var c model.Content
		err := rows.Scan(&c.ID, &c.URL, &c.Title, &c.Summary, &c.Category, 
		                 pq.Array(&c.Tags), &c.Status, &c.CreatedAt)
		if err != nil {
			return nil, err
		}
		contents = append(contents, c)
	}
	return contents, nil
}
```

package service

import (
    "github.com/your-username/zhizhou/server/internal/model"
    "github.com/your-username/zhizhou/server/internal/repository"
    "github.com/your-username/zhizhou/server/internal/pkg/crypto"
)

type APIKeyService interface {
    SaveAPIKey(userID, provider, plainKey, baseURL, modelName string) error
    ListAPIKeys(userID string) ([]model.APIKey, error)
}

type apiKeyService struct {
    repo repository.APIKeyRepository
    encKey []byte
}

func NewAPIKeyService(repo repository.APIKeyRepository, encKey string) APIKeyService {
    return &apiKeyService{
        repo: repo,
        encKey: []byte(encKey),
    }
}

func (s *apiKeyService) SaveAPIKey(userID, provider, plainKey, baseURL, modelName string) error {
    encrypted, err := crypto.Encrypt(s.encKey, plainKey)
    if err != nil {
        return err
    }
    k := &model.APIKey{
        UserID:          userID,
        Provider:        provider,
        APIKeyEncrypted: encrypted,
        BaseURL:         baseURL,
        Model:           modelName,
    }
    return s.repo.Create(k)
}

func (s *apiKeyService) ListAPIKeys(userID string) ([]model.APIKey, error) {
    return s.repo.GetByUserID(userID)
}

package handler

import (
    "net/http"
    "github.com/gin-gonic/gin"
    "github.com/your-username/zhizhou/server/internal/service"
)

type APIKeyHandler struct {
    service service.APIKeyService
}

func NewAPIKeyHandler(s service.APIKeyService) *APIKeyHandler {
    return &APIKeyHandler{service: s}
}

func (h *APIKeyHandler) Save(c *gin.Context) {
    var req struct {
        Provider string `json:"provider" binding:"required"`
        Key      string `json:"key" binding:"required"`
        BaseURL  string `json:"base_url"`
        Model    string `json:"model" binding:"required"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }
    
    // 从 middleware context 获取 userID
    userID := c.GetString("userID") 
    
    err := h.service.SaveAPIKey(userID, req.Provider, req.Key, req.BaseURL, req.Model)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "failed to save key"})
        return
    }
    c.JSON(http.StatusOK, gin.H{"message": "success"})
}

## 11. 网页解析 ✅

路径：`server/internal/pkg/parser/web.go`

```go
package parser

import (
    "bytes"
    "net/http"
    "net/url"
    "time"

    readability "codeberg.org/readeck/go-readability/v2"
)

// ParseWebContent 提取网页正文
// 国内适配：设置了 User-Agent 和超时，避免被 CSDN/掘金/公众号等站点拦截
func ParseWebContent(rawURL string) (string, string, error) {
    client := &http.Client{Timeout: 30 * time.Second}

    req, err := http.NewRequest("GET", rawURL, nil)
    if err != nil {
        return "", "", err
    }
    req.Header.Set("User-Agent", "Mozilla/5.0 (compatible; ZhiZhou/1.0; +https://zhizhou.app)")

    resp, err := client.Do(req)
    if err != nil {
        return "", "", err
    }
    defer resp.Body.Close()

    parsedURL, err := url.Parse(rawURL)
    if err != nil {
        return "", "", err
    }

    article, err := readability.FromReader(resp.Body, parsedURL)
    if err != nil {
        return "", "", err
    }

    // v2 API：Title() 是方法不是字段；TextContent 已删除，改用 RenderText
    var buf bytes.Buffer
    if err := article.RenderText(&buf); err != nil {
        return article.Title(), "", nil // 标题能拿到，文本提取失败不阻断
    }

    return article.Title(), buf.String(), nil
}
```

> ⚠️ **已修正 4 处**：
> 1. 库路径：已归档的 `github.com/go-shiori/go-readability` → `codeberg.org/readeck/go-readability/v2`
> 2. `FromReader` 第二个参数：`nil`（运行时 panic）→ `*url.URL` 对象
> 3. **v2 重大变更**：`article.TextContent`（v1 字段）在 v2 中**已删除**，需用 `article.RenderText(&buf)` 获取文本
> 4. **v2 重大变更**：`article.Title`（v1 字段）在 v2 中**变为方法** `article.Title()`，不加括号编译通过但返回的是函数不是字符串
> 5. 添加了 `http.Client` 超时 + User-Agent（国内站点兼容）
> 6. 补上了 `"bytes"` 和 `"time"` 导入

                placeholder="粘贴文章链接..."
                required
            />
            <button disabled={loading}>
                {loading ? '处理中...' : '采集'}
            </button>
        </form>
    );
};

## 19. 🔴 国内适配清单（Agent 忽略的关键问题）

前一个 Agent（Gemini）默认以西方/OpenAI 生态为基准，以下问题已逐一修正。

| # | 问题 | Agent 原文 | 修正后 | 影响 |
|---|------|-----------|--------|------|
| 1 | **go-readability 已归档** | `github.com/go-shiori/go-readability`（2025-12-30 归档） | `codeberg.org/readeck/go-readability/v2` | 不换则无法获取依赖更新 |
| 2 | **伪版本号（幻觉）** | `v0.0.0-20241001...`（Gemini 编造，不存在） | `v2.1.1`（最新稳定版） | go mod tidy 会失败 |
| 3 | **FromReader 传 nil** | `readability.FromReader(resp.Body, nil)` | 传入 `*url.URL` 对象 | 运行时 panic |
| 4 | **v2 API 重大变更：Title 变方法** | `article.Title`（v1 字段） | `article.Title()`（v2 方法） | 编译通过但返回函数不是字符串 |
| 5 | **v2 API 重大变更：TextContent 已删除** | `article.TextContent`（v1 才有的字段） | `article.RenderText(&buf)` | 编译失败，字段不存在 |
| 6 | **VECTOR(1536) 硬编码** | `embedding VECTOR(1536)` 注释写"适配 OpenAI" | 环境变量 `EMBEDDING_DIM` 控制 | 切换 embedding 模型需改表结构 |
| 7 | **pq.Array 缺失** | `&c.Tags` 直接扫描 TEXT[] | `pq.Array(&c.Tags)` 包装 | 运行时 `unsupported Scan` 错误 |
| 8 | **http.Get 无 UA/超时** | 裸 `http.Get(url)` | `http.Client` + 30s 超时 + 知舟 UA | CSDN/掘金/公众号拒绝 Go 默认 UA |
| 9 | **缺少导入** | 缺 `"net/url"` `"time"` `"bytes"` | 已补全 | 编译失败 |
| 10 | **title 硬编码** | `"正在处理中..."` | `c.Title`（从 parser 获取真实标题） | 所有入库内容标题都是"正在处理中..." |

### 后续开发中需注意的国内合规项

| 项 | 说明 |
|----|------|
| 短信验证码 | 用阿里云短信/腾讯云短信，不应用 Twilio 等国外服务 |

| 支付 | 微信支付 / 支付宝，不应用 Stripe |
| 部署 | 支持阿里云/腾讯云/华为云，Docker 镜像源用国内镜像 |
